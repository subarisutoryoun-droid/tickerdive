  <script>

    // ── URL params → fill fields ──

    const params = new URLSearchParams(location.search);

    const set = (id, val) => { if (val) document.getElementById(id).textContent = val; };


    set('eventName',  params.get('event_name') || 'EVENT NAME');

    set('holderName', params.get('name')        ? params.get('name') + ' さま' : 'ゲスト さま');

    set('seatNo',     params.get('seat')        || 'SA-0');

    set('venue',      params.get('venue')       || '会場未定');

    set('eventDate',  params.get('date')        || '—');

    set('openTime',   'OPEN ' + (params.get('open')  || '—'));

    set('startTime',  'START ' + (params.get('start') || '—'));


    const ticketId = params.get('event') || 'UNKNOWN';

    document.getElementById('ticketId').textContent = 'TICKET ID: ' + ticketId.toUpperCase();


    // ── Simple QR placeholder (uses Google Charts API) ──

    const qrUrl = `https://chart.googleapis.com/chart?chs=120x120&cht=qr&chl=${encodeURIComponent(location.href)}&choe=UTF-8`;

    const img = document.createElement('img');

    img.src = qrUrl;

    img.style.width = '100%';

    img.style.height = '100%';

    img.style.borderRadius = '4px';

    document.getElementById('qrContainer').appendChild(img);
